打开一个 Confirm 弹出层，成功时计数加 1

CONFIRM
计数：0

nicePopup.confirm.open('是否计数加 1 ？', function (popup) {
  popup.close()
  count()
})
function count () {
  if (!this.countNum) {this.countNum = 0}
  this.countNum++
  document.getElementById('count').innerHTML = '计数：' + this.countNum
}
